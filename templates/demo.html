使用示例

1. 基础登录

// 检查登录状态 if (window.authService.isLoggedIn()) { console.log('用户已登录:', window.authService.getCurrentUser()); } else { console.log('用户未登录'); }

2. 发起登录

// Auth0登录（推荐） await window.authService.auth0Login(); // Google直接登录 await window.authService.googleLogin(); // 微信登录 await window.authService.wechatLogin();

3. API调用

// 带认证的API调用 try { const response = await window.authService.authenticatedRequest('/api/some-endpoint', { method: 'POST', body: JSON.stringify({ data: 'example' }) }); const result = await response.json(); console.log('API响应:', result); } catch (error) { console.error('API调用失败:', error); }

4. 登出

// 登出 window.authService.logout();
